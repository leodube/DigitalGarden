```typescript
let openParenth = ['(', '{', '['];
let closedParenth = [')', '}', ']'];

function isValid(s: string): boolean {
    let sa = [];    
    for (let i=0; i<s.length; i++) {
        if (openParenth.includes(s[i])) {
            sa.push(s[i]);
        } else if (closedParenth.includes(s[i])) {
            switch (s[i]) {
                case ')':
                    if (sa[sa.length-1] == '(') 
                        sa.pop();
                    else
                        return false;
                    break;
                case '}':
                    if (sa[sa.length-1] == '{') 
                        sa.pop();
                    else
                        return false;
                    break;
                case ']':
                    if (sa[sa.length-1] == '[') 
                        sa.pop();
                    else
                        return false;
                    break;
                default:
                    return false;
            }
        } else {
            return false;
        }
    }
    
    return sa.length == 0 ? true : false
};
```

```typescript
function isValid(s: string): boolean {
    const o = {
        ')':'(',
        '}':'{',
        ']':'['
    };
    
    const sa = [];
    
    for (let i = 0; i < s.length; i++) {
        let currB = s[i];
        
        if (!(currB in o)){
            sa.push(currB);
        } else if (sa.at(-1) === o[currB]) {
            sa.pop();
        } else {
            return false;
        }
    }
    
    return sa.length ? false : true;
};
```
